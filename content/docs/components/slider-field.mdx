## Usage

```tsx
import SliderField from "@/components/slider-field";
```

```tsx
<SliderField
  name="volume"
  label="Volume"
  description="Adjust the volume level"
  min={0}
  max={100}
  step={1}
/>
```

## Usage Slider range

```tsx
<SliderField
  name="volume"
  label="Volume"
  description="Adjust the volume level"
  min={0}
  max={100}
  step={1}
  defaultValue={[20, 80]}
/>
```
