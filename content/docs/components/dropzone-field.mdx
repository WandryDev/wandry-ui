## Usage

```tsx
import DropzoneField from "@/components/dropzone-field";
```

```tsx
<DropzoneField
  name="files"
  label="Upload Files"
  description="Drag and drop files here or click to select files."
  accept={{ "image/*": [], "application/pdf": [] }}
  maxFiles={5}
  maxSize={5 * 1024 * 1024} // 5 MB
/>
```
